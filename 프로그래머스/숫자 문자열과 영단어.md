class Solution {
    public int solution(String s) {
        int answer = 0;
        
        String[][] changeMap = {{"zero", "0"}, {"one", "1"}, {"two", "2"}, {"three", "3"}, {"four", "4"}
                         , {"five", "5"}, {"six", "6"}, {"seven", "7"}, {"eight", "8"}, {"nine", "9"}};

        for(String[] item: changeMap){
            s = s.replace(item[0], item[1]);
        }
        
        answer = Integer.parseInt(s);
        
        return answer;
    }
}
