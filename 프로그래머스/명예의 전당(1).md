import java.util.Arrays;

class Solution {
    public int[] solution(int k, int[] score) {
        int[] answer = new int[score.length];
        
        int[] honor = Arrays.copyOf(score, k);
        Arrays.sort(honor);
        Arrays.fill(answer, honor[0]);
        
        if(k<score.length){
            for(int i = k ; i < score.length; i++){
            if(honor[0] < score[i]){
                honor[0] = score[i];
                Arrays.sort(honor);
            }
            answer[i] = honor[0];
            }
        }
        
        return answer;
    }
}
